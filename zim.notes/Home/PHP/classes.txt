====== classes ======

**Creating classes:**

class Comment {
    public __static function__ add($author, $body, $article_id) {								=> static functions (class functions)
        $addCommentSQL = "INSERT INTO comments (id, author, body, article_id) " .
                         "VALUES (NULL, '{$author}', '{$body}', '{$article_id}');";
        $results = mysql_query($addCommentSQL) or die ("error in SQL query {$addCommentSQL}");
        return true;
    }

    public $id;								=> __properties must have an explicit visibility__ (unless used w/ var, don't do this)
    public $author;
    public $body;

    function ____construct__($id, $author, $body) {
        $this->id = $id;
        $this->author = $author;
        $this->body = $body;
    }
}
